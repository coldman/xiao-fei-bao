Class SortView

用于显示排序按钮/链接的控件, 并包含 js 排序功能. 这些排序控件附加在指定的 HTMLElement 里.

SortView()

Author: ideawu@163.com. 
var rows = []; // datasource

var sort = new SortView();
sort.fields = {
	'id'	: ['int', null],
	'name'	: [null, null]
};

var elements = {};
var ths = $('table th').each(function(i, th){
	var k = $(th).attr('field');
	if(k != undefined){
		elements[k] = th;
	}
});
sort.onclick = function(){
	sort.sort(rows);
	// do with rows...
}

sort.render(elements);
